//Date from count-down

var grad_day = new Date("May 28, 2008 00:00:00 GMT");

var box = null;

var tempX = 0, tempY = 0;

var boxX = 0, boxY = 0;

var IE = document.all?true:false;


//Compatibility taken from //http://www.thescripts.com/forum/thread150939.html

if (!IE) document.captureEvents(Event.MOUSEMOVE)

function getMouseXY(e) {

  if (IE) { // grab the x-y pos.s if browser is IE

    tempX = event.clientX + document.body.scrollLeft

    tempY = event.clientY + document.body.scrollTop

  } else {  // grab the x-y pos.s if browser is NS

    tempX = e.pageX

    tempY = e.pageY

  }

  if (tempX < 0){tempX = 0}

  if (tempY < 0){tempY = 0}  

  return true

}


//Box

function doHeight(){

  box.style.height =(Math.round(100 - Math.sqrt(

  Math.pow (tempX - boxX, 2) +

  Math.pow (tempY - boxY, 2) ))).toString();

  setTimeout(doHeight,1);

}


function doMove() {

  boxX = tempX;

  boxY = tempY;

  box.style.left = tempX; //parseInt(box.style.left)+1+'px';

  box.style.top = tempY;

  setTimeout(doMove,200); //call doMove in 20msec

}


function doTimer() {

  var cur_day = new Date()

  var time_left = Math.floor(

                 (grad_day.getTime()-cur_day.getTime()) /1000);


  var secs = time_left %60;

  time_left=Math.floor(time_left/60);


  var mins = time_left%60;

  time_left = Math.floor(time_left/60);


  var hours = time_left%24;

  time_left = Math.floor(time_left/24);

  

  var days = time_left;   

 

  var output = "Freedom in: "+ "Days: " + days + " Hours: " +

                hours + " Mins: " + mins + " Secs: " + secs;

  document.getElementById('Box').innerHTML= output;

  setTimeout(doTimer,100);

}


function init() {

  box = document.getElementById('Box');

  box.style.left = '0px';

  doMove(); 

  doHeight();

  doTimer();

}



window.onload = init;
